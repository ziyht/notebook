## 创建服务

```sh
#!/bin/sh

dir=$(cd $(dirname $0); pwd)

# 在这里进行相关配置
name=test_service                                   # <= 设置服务名称
cmd="/usr/bin/python $dir/src/app.py --dev"         # <= 设置执行命令，必须是绝对路径
workdir=$dir                                        # <= 设置工作目录

help=`cat <<EOF
  
./service <install,uninstall,reinstall,enable,disable,start,stop,restart,status,logs>
  
  install      install service, will set enable automatically
  uninstall    uninstall service
  reinstall    exec uninstall and install
  enable       set enable so it will load on startup
  disable      set disable and it will not load on startup 
  start        start the service
  stop         stop the service
  restart      restart the service
  status       show the status of current service
  logs         show logs in outputs of current service

EOF`

sfile=/etc/systemd/system/$name.service

install(){
    if [[ -a $sfile ]];then
        echo service $name already installed
        exit 0
    fi

    echo "install..."
    cat << EOF > $sfile
[Unit]
Description=$name
 
[Service]
Type=simple
WorkingDirectory=$workdir
ExecStart=$cmd
Restart=always
RestartSec=5
StartLimitInterval=3
RestartPreventExitStatus=137
 
[Install]
WantedBy=multi-user.target
EOF
    systemctl enable $name
    echo "install over"
}

start() {
    if [[ -a $sfile ]];then
        echo starting $name ...
        systemctl start $name
    else
        echo please install first
        exit 1
    fi
}

stop() {
    if [[ -a $sfile ]];then
        echo stopping $name ...
        systemctl stop $name
    else
        echo please install first
    fi 
}

uninstall(){
    if [[ -a $sfile ]];then
        echo uninstall $name
        rm -f $sfile
    fi

    echo uninstalled $name
}

status() {
    if [[ -a $sfile ]];then
        systemctl status $name
    else
        echo not installed yet
    fi
}

enable() {
    if [[ ! -a $sfile ]];then
        install
    fi

    echo enable $name
    systemctl enable $name
}

disable() {
    if [[ -a $sfile ]];then
        echo disable $name
        systemctl disable $name
    else
        echo not installed yet
    fi
}

logs() {
    if [[ -a $sfile ]];then
        journalctl -f -e -u $name
    else
        echo not installed yet
    fi
}

if [[ $# -ne 1 ]]; then
    echo $help
    exit 0
fi

if [[ $1 == "install" ]]; then
    install
    enable
    exit 0
fi

if [[ $1 == "uninstall" ]]; then
    stop
    disable
    uninstall
    exit 0
fi

if [[ $1 == "reinstall" ]]; then
    stop
    disable
    uninstall
    install
    enable
    exit 0
fi

if [[ $1 == "start" ]]; then
    start
    status
    exit 0
fi

if [[ $1 == "stop" ]]; then
    stop
    status
    exit 0
fi

if [[ $1 == "restart" ]]; then
    stop
    start
    status
    exit 0
fi

if [[ $1 == "status" ]]; then
    status
    exit 0
fi

if [[ $1 == "enable" ]]; then
    enable
    exit 0
fi

if [[ $1 == "disable" ]]; then
    disable
    exit 0
fi

if [[ $1 == "logs" ]]; then
    logs
    exit 0
fi

echo "$help"
```
